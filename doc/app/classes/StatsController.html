49:   def forum_stats
50:       g = Gruff::Line.new('580x210')
51:       g.theme = {
52:          :colors => ['#ff6600', '#3bb000', '#1e90ff', '#efba00', '#0aaafd'],
53:          :marker_color => '#aaaaaa',
54:          :background_colors => ['#eaeaea', '#ffffff']
55:        }
56: 
57:       g.hide_title = true
58:       # g.hide_legend = true
59:       g.font = File.expand_path('fonts/Vera.ttf', RAILS_ROOT)
60:       
61:       range = "created_at #{(30.days.ago.to_date...Date.today+1.day).to_s(:db)}"
62:       @posts = @semester.forums.find_discussion.posts.count(:all, :conditions => range,
63:               :group => "DATE_FORMAT(created_at, '%m-%d')", :order =>"created_at ASC")
64:                 
65:       @gallery_posts = @semester.forums.find_photo.posts.count(:all, :conditions => range, 
66:               :group => "DATE_FORMAT(created_at, '%m-%d')", :order =>"created_at ASC")
67: 
68:       # Take the union of all keys & convert into a hash {1 => "month", 2 => "month2"...}
69:       # - This will be the x-axis.. representing the date range
70:       days = (@posts.keys | @gallery_posts.keys).sort
71:       keys = Hash[*days.collect {|v| [days.index(v),v.to_s] }.flatten]
72: 
73:       # Plot the data - insert 0's for missing keys
74:       g.data("Discussion", keys.collect {|k,v| @posts[v].nil? ? 0 : @posts[v]})
75:       g.data("Gallery", keys.collect {|k,v| @gallery_posts[v].nil? ? 0 : @gallery_posts[v]})
76:       
77:       g.labels = keys
78:       # g.x_axis_label = nil
79:       send_data(g.to_blob, :disposition => 'inline', :type => 'image/png', :filename => "forum-stats.png")
80:     end
