53:   def distance_of_time_in_words(from_time, to_time = 0, include_seconds = false)
54:     from_time = from_time.to_time if from_time.respond_to?(:to_time)
55:     to_time = to_time.to_time if to_time.respond_to?(:to_time)
56:     distance_in_minutes = (((to_time - from_time).abs)/60).round
57:   
58:     case distance_in_minutes
59:       when 0..1           then (distance_in_minutes==0) ? 'a few seconds ago' : '1 minute ago'
60:       when 2..59          then "#{distance_in_minutes} minutes ago"
61:       when 60..90         then "1 hour ago"
62:       when 90..1440       then "#{(distance_in_minutes.to_f / 60.0).round} hours ago"
63:       when 1440..2160     then '1 day ago' # 1 day to 1.5 days
64:       when 2160..2880     then "#{(distance_in_minutes.to_f / 1440.0).round} days ago" # 1.5 days to 2 days
65:       else from_time.strftime("%b %e, %Y  %l:%M%p").gsub(/([AP]M)/) { |x| x.downcase }
66:     end
67:   end
